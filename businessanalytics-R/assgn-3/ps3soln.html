$$Pr[approve]=\beta_0+\beta_1white+\beta_2obrat+\beta_3white\times obrat$$

$$\frac{dPr[approve]}{dwhite}=\beta_1+\beta_3obrat$$

$$\theta_1=\beta_1+32\beta_3$$ $$\theta_1-32\beta_3=\beta_1$$

$$Pr[approve]=\beta_0+(\theta_1-32\beta_3)white+\beta_2obrat+\beta_3white\times obrat$$ $$Pr[approve]=\beta_0+\theta_1white+\beta_2obrat+\beta_3white\times obrat-32\beta_3white$$ $$Pr[approve]=\beta_0+\theta_1white+\beta_2obrat+\beta_3white(obrat-32)$$

rep.kpss <- function(series,alpha=0.05,dmax=5){
  diff <- 0
  for(i in 1:dmax){
    suppressWarnings(pval <- kpss.test(series,null="Level")$p.value)
    if(pval>=alpha){
      return(c(diff,0,pval))
    }
    suppressWarnings(pval <- kpss.test(series,null="Trend")$p.value)
    if(pval>=alpha){
      return(c(diff,1,pval))
    }
    diff <- diff + 1
    series <- diff(series)
  }
  return(NULL)
}
